Database test 1


select doctor_name , patient_name,  concat(`Slots`.`date` ,' ', `Slots`.`time`) as slot_date_time  


from Appointments left join `Slots` on `Appointments`.`slot_id` = `Slots`.`id`


left join  Doctors on  `Appointments`.`doctor_id` = `Doctors`.id 


where MONTH( `Slots`.`date`) = 10


order by slot_date_time ASC



Db 2



select doctor_name , count( Appointments.`doctor_id` )  as appointment_count  From Appointments


left join  Doctors on  `Appointments`.`doctor_id` = `Doctors`.id 


group by doctor_id limit 1



DB 3



select  Doctors.doctor_name , SUM( `Slots`.`duration` ) as total_duration


from Appointments left join `Slots` on `Appointments`.`slot_id` = `Slots`.`id`


left join  Doctors on  `Appointments`.`doctor_id` = `Doctors`.id 


group by Doctors.`id`


order by total_duration DESC
